import { Component, EventEmitter, Input, Output } from '@angular/core';

@Component({
  selector: 'authentication-button',
  templateUrl: './authentication-button.component.html',
  styleUrl: './authentication-button.component.css'
})
export class AuthenticationButtonComponent {
  @Input() buttonText: string = "";
  @Input() buttonType: string = "";
  @Input() icon: string = "";
  @Output() buttonClick = new EventEmitter<void>();

  constructor() { }

  onClick(): void {
    this.buttonClick.emit();
  }
}


<button (click)="onClick()" [type]="buttonType" class="btn-with-icon">
    <img *ngIf="icon" [src]="icon" class="button-icon" />
    <span>{{ buttonText }}</span>
 </button>
